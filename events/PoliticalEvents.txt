add_namespace = political

# Technicism on the Rise
country_event = {
	id = political.1
	title = political.1.t
	desc = political.1.d
	picture = GFX_report_event_fascist_gathering

	trigger = {
		NOT = { has_government = technicism }
		technicism > 0.6
	}

	fire_only_once = yes

	mean_time_to_happen = {
		days = 30
	}

	# We need a referendum
	option = {
		name = political.1.a
		ai_chance = {
			factor = 1
		}
		add_political_power = -100
		set_politics = {
			ruling_party = technicism
			elections_allowed = no
		}
	}

	# Not right now
	option = {
		name = political.1.b
		ai_chance = {
			factor = 1
		}
		custom_effect_tooltip = impending_civil_war_tt
		set_country_flag = technicism_popular_movement
	}
}


# Civil War for technicism
country_event = {
	id = political.2
	title = political.2.t
	desc = political.2.d
	picture = GFX_report_event_fascist_gathering

	trigger = {
		has_idea_with_trait = technicist_reformer
		NOT = { has_government = technicism }
		technicism > 0.25
		OR = {has_country_flag = technicism_popular_movement}
	}

	mean_time_to_happen = {
		days = 365
		modifier = {
			factor = 0.75
			has_country_flag = technicism_popular_movement
		}
	}

	option = {
		name = political.2.a
				start_civil_war = {
				ruling_party = technicism
				ideology = ROOT
					size = 0.6
				}
	}
	option = {
		name = political.2.b
		start_civil_war = {
		ideology = technicism
			size = 0.6
		}
	}
	}
