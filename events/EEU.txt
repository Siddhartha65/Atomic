    }

	days = 1

		name = eeu.1.a

		DDR = { white_peace = GER }
		DDR = { white_peace = DEN }

		create_country_leader = {
		name = "Christoph Josten"
		desc = "POLITICS_MOHAMMED_ZAHIR_SHAH_DESC"

	}
}
complete_national_focus = DDR_president_peace
}
